<i><font color="#b22222">/*
 * This little program checks whether your operating system allows a process
 * to attach to another process whoose uid is only identical because it
 * revoked its privileges.
 * 
 * If it detects that your operating system may be unsafe, then it's probably
 * better to avoid usage of privilege separation if untrusted users have
 * shell access.
 *
 * Compile and run with :
 * 
 * make ptracetest
 * ./ptracetest

#<b><font color="#5f9ea0">include</font></b> <font color="#bc8f8f"><b>&lt;config.h&gt;</b></font>

#<b><font color="#5f9ea0">if</font></b> !<b><font color="#5f9ea0">defined</font></b>(<font color="#b8860b">HAVE_PTRACE</font>) || !<b><font color="#5f9ea0">defined</font></b>(<font color="#b8860b">HAVE_SYS_PTRACE_H</font>)

    fputs(<font color="#bc8f8f"><b>"Sorry, this test can't be compiled in this platform\n"</b></font>, stderr);

<font color="#228b22"><b>int</b></font> <b><font color="#0000ff">main</font></b>(<font color="#228b22"><b>void</b></font>)
{
    pid_t pid;
    <font color="#228b22"><b>int</b></font> rtn = 1;

    <b><font color="#a020f0">if</font></b> (geteuid() != (uid_t) 0) {
        fputs(<font color="#bc8f8f"><b>"Sorry, you need to run this program as root\n"</b></font>, stderr);
        <b><font color="#a020f0">return</font></b> 254;
    }
    <b><font color="#a020f0">if</font></b> (setgid((gid_t) TEST_GID) || setegid((gid_t) TEST_GID) ||
        setuid((uid_t) TEST_UID) || seteuid((uid_t) TEST_UID)) {
        perror(<font color="#bc8f8f"><b>"Error while switching gid/uid"</b></font>);
        <b><font color="#a020f0">return</font></b> 254;
    }
    <b><font color="#a020f0">if</font></b> ((pid = fork()) == (pid_t) -1) {
        perror(<font color="#bc8f8f"><b>"Unable to fork"</b></font>);
        <b><font color="#a020f0">return</font></b> 254;
    }
    <b><font color="#a020f0">if</font></b> (pid == (pid_t) 0) {
        size_t t = (size_t) 0U;

        fputs(<font color="#bc8f8f"><b>"Checking for traceability after uid change "</b></font>, stdout);
        fflush(stdout);
        <b><font color="#a020f0">do</font></b> {
            putchar(ZIPPER[t]);
            putchar(<font color="#bc8f8f"><b>'\b'</b></font>);
            fflush(stdout);
            (<font color="#228b22"><b>void</b></font>) sleep(1U);
            t++;
        } <b><font color="#a020f0">while</font></b> (t &lt; <b><font color="#a020f0">sizeof</font></b> ZIPPER - (size_t) 1U);

        _exit(0);
    } <b><font color="#a020f0">else</font></b> {
        <font color="#228b22"><b>int</b></font> status;
        <font color="#228b22"><b>long</b></font> ret;        

# <b><font color="#5f9ea0">ifdef</font></b> <font color="#b8860b">PT_ATTACH</font>
        ret = ptrace(PT_ATTACH, pid, NULL, NULL);
# <b><font color="#5f9ea0">else</font></b>
        ret = ptrace(PTRACE_ATTACH, pid, NULL, NULL);        
# <b><font color="#5f9ea0">endif</font></b>
        
        <b><font color="#a020f0">while</font></b> (wait(&amp;status) != pid);

                 <font color="#bc8f8f"><b>"users don't have shell access."</b></font>);
        }        
    }
